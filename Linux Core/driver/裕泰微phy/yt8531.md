```
#define REG_DEBUG_ADDR_OFFSET          0x1e
#define REG_DEBUG_DATA                 0x1f
static int ytphy_mii_rd_ext(struct mii_bus *bus, int phy_id, u32 regnum)
{
	int ret;
	int val;

	ret = bus->write(bus, phy_id, REG_DEBUG_ADDR_OFFSET, regnum);
	if (ret < 0)
		return ret;

	val = bus->read(bus, phy_id, REG_DEBUG_DATA);

	return val;
}

static int ytphy_mii_wr_ext(struct mii_bus *bus, int phy_id, u32 regnum, u16 val)
{
	int ret;

	ret = bus->write(bus, phy_id, REG_DEBUG_ADDR_OFFSET, regnum);
	if (ret < 0)
		return ret;

	ret = bus->write(bus, phy_id, REG_DEBUG_DATA, val);

	return ret;
}
```

```
#!/bin/bash

# 脚本功能：前两位十六进制按二进制位步进（每次+4），后两位从0xF1到0xFF循环
# 完整逻辑：0x00F1 → 0x04F1 → 0x08F1 → ... → 0x3CF1 → 0x00F2 → ... → 0x3CFF

# 定义起始和终止地址（必须为4位十六进制）
start_addr="0x00FE"
end_addr="0x3CFF"

# 提取地址各部分（前两位+后两位）
start_prefix="${start_addr:2:2}"      # 起始前两位（00）
start_suffix="${start_addr:4:2}"      # 起始后两位（F1）
end_prefix="${end_addr:2:2}"          # 终止前两位（3C）
end_suffix="${end_addr:4:2}"          # 终止后两位（FF）

# 转换为十进制
current_prefix=$((16#$start_prefix))  # 00 → 0
end_prefix_decimal=$((16#$end_prefix)) # 3C → 60
current_suffix=$((16#$start_suffix))  # F1 → 241
end_suffix_decimal=$((16#$end_suffix)) # FF → 255

# 验证地址范围
if [[ $current_prefix -gt $end_prefix_decimal || $current_suffix -gt $end_suffix_decimal ]]; then
  echo "错误：起始地址不能大于终止地址！"
  exit 1
fi

# 主循环
while [ $current_suffix -le $end_suffix_decimal ]; do
  # 重置前两位为起始值（每次后两位进位后前两位重新开始）
  current_prefix=$((16#$start_prefix))
  
  # 前两位按步长4递增（二进制位步进）
  while [ $current_prefix -le $end_prefix_decimal ]; do
    # 格式化当前地址
    hex_prefix=$(printf "%02X" $current_prefix)
    hex_suffix=$(printf "%02X" $current_suffix)
    full_addr="0x${hex_prefix}${hex_suffix}"
    
    # 输出当前地址（可替换为实际操作）
    # echo "[$(date '+%Y-%m-%d %H:%M:%S')] 当前地址: $full_addr"
    /home/root/phyextwrite eth1 0x1 0xA003 ${full_addr}
    iperf3 -c 192.168.1.100 -i 1 -t 10 -b 1000M

    # 步进并等待（最后一次循环不等待）
    current_prefix=$((current_prefix + 4))
    if [ $current_prefix -le $end_prefix_decimal ]; then
      sleep 10
    fi
  done
  
  # 后两位递增
  current_suffix=$((current_suffix + 1))
done

echo "循环完成。"
```

## SGMAC_TO_RGPHY-mode
连接方式是
CPU --(RGMII)--> YT8531S --(SGMII)--> sgmii mac芯片
平台以stm32mp257为例
读0xA001寄存器，把最后一位改成0x5，同时还要禁用自协商和强制1000M


修改drivers/net/phy/motorcomm.c
添加
```
static int yt8531S_sgmii_mac_init(struct phy_device *phydev)
{
	int ret = 0;
	u16 tmp_val;
	struct device_node *np;

	printk("yt8531S: eth name = %s", phydev->attached_dev->name);
	if (strcmp("end2", phydev->attached_dev->name) == 0)
	{
		np = phydev->mdio.dev.of_node;
		if (!np ){
			netdev_info(phydev->attached_dev,"No platform data");
			return -ENODEV;
		}
		/* SGMAC_TO_RGPHY */
		tmp_val = ytphy_read_ext(phydev, YT8521_CHIP_CONFIG_REG);
		printk("yt8531S: defautl 0xA001 val is %x", tmp_val);
		tmp_val = (tmp_val & ~0x7) | 0x5; //bit[2:0] : 101 SGMAC_TO_RGPHY
		ret = ytphy_write_ext(phydev, YT8521_CHIP_CONFIG_REG, tmp_val);
		if (ret < 0) {
			netdev_info(phydev->attached_dev, "failed to write 0xa001");
			return ret;
		}
		tmp_val = ytphy_read_ext(phydev, YT8521_CHIP_CONFIG_REG);
		printk("yt8531S: readback 0xA001 val is %x", tmp_val);
		
		/* speed 1000Mb, disable auto-neg */
		/*  it will be configured by yt8521_config_aneg */
		phydev->autoneg = AUTONEG_DISABLE;
		phydev->speed = SPEED_1000;
		phydev->duplex = DUPLEX_FULL;
		phydev_info(phydev, "end2: forced 1000M full-duplex\n");
		
	}
	return ret;
}
```
